```twig
{# 注释 #}  
  
{# 打印内容 #}  
{{ '不使用双引号？' }}  
  
{# 声明、设置变量 #}  
{% set a = 1 %}  
{% set a,b = 1,2 %}  
  
{# 拼接字符串用 ~ #}{{ 'a = ' ~ a ~ ' and b = ' ~ b }}  
  
{#声明、设置一维数组#}  
{% set arr = [1,2,3] %}  
{{ arr[0] }}  
{{ arr.0 }}  
  
{#关联数组#}  
{% set arr2 = ['one' => 1] %}  
  
 {#对象#}  
{% set foo = { 'foo' : 'bar' } %}  
{{ foo.foo }}  
  
{#当属性包含特殊字符时（比如-，将被解析为减号操作符），使用attribute函数访问变量属性：#}  
{% set foo = { 'ab-cd':100 } %}  
{{ attribute(foo, 'data-foo') }}  
  
{#过滤器，用 | 分割变量和函数  
前面是要处理的变量，后面是过滤器名字，也就是函数了#}  
{% set arr = ['a','b','c'] %}  
{{ arr|join(',')|upper }}{#链式、带参数的过滤器#}  
{% filter upper %} {#过滤器名字：upper#}  
  {{ arr|join(',') }}{#带参数的过滤器#}  
{% endfilter %}  
  
{#for循环会默认加一#}
```
